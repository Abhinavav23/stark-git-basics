initialize a git repository
-- git init

connect your local with remote repo
-- git remote add origin "origin path"

move files to staging area
-- git add "file path"
-- git add . --> will add all the files to staging area

checking the status of the files
-- git status

commit a file
-- git commit -m "message"

pushing code to remote repo
-- git push origin master

branching in git 
-- git branch --> show the list of branches
-- git branch [branch name] --> create a new branch
-- git checkout [branch name] --> to move to that branch
